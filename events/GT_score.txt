#Trigger event
country_event = {
    id = GT_score.2
    title = GT_score.2.t
    is_triggered_only = yes

    trigger = {
        always = yes
    }

    hidden = yes

    immediate = {
        every_country = {
            limit = {
                has_country_flag = gt_campaign_nation
            }
            if = {
                limit = {
                    has_country_flag = gt_bishop
                }
                country_event = {
                    id = GT_score.3
                }
            }
            if = {
                limit = {
                    has_country_flag = gt_prince
                }
                country_event = {
                    id = GT_score.4
                }
            }
            if = {
                limit = {
                    has_country_flag = gt_magnate
                }
                country_event = {
                    id = GT_score.5
                }
            }
            if = {
                limit = {
                    has_country_flag = gt_despot
                }
                country_event = {
                    id = GT_score.6
                }
            }
        }
    }
    option = {
        name = GT_score.2.a
    }
}

#Bishop score
country_event = {
    id = GT_score.3
    title = GT_score.3.t
    desc = GT_score.3.d
    is_triggered_only = yes
    hidden = yes
    trigger = {
        always = yes
    }

    immediate = {
        set_variable = {            
            which = gt_gained_score
            value = 0
        }
        #20 points for each Conqueror Objective or Imperial Objective province that is Catholic (cooperative objective).
        #Hamburg
        every_province = {
            limit = {
                OR = {
                    has_province_flag = gt_imperial_objective
                    has_province_flag = gt_conqueror_objective
                }
                religion = catholic
            }
            ROOT = {
                change_variable = {
                    which = gt_gained_score
                    value = 20
                }
            }
        }
        #100 points for holding the Curia.
        if = {
            limit = {
                is_papal_controller = yes
            }
            change_variable = {            
                which = gt_gained_score
                value = 100
            }
        }
        #30 points for each alliance with an Elector, the Holy Roman Emperor, or the Catholic Defender of the Faith, as long as that nation is not a member of the Bishop group.
        every_ally = {
            limit = {
                AND = {
                    NOT = { has_country_flag = gt_bishop }
                    OR = {
                        is_elector = yes
                        is_emperor = yes
                        AND = {
                            religion = catholic
                            is_defender_of_faith = yes
                        }
                    }
                }
            }
            ROOT = {
                change_variable = {            
                    which = gt_gained_score
                    value = 30
                }
            }
        }
        #20 x (DotF Rank) pts for holding a Defender of the Faith title.
        if = {
            limit = {
                is_defender_of_faith_of_tier = 5
            }
            change_variable = {            
                which = gt_gained_score
                value = 100
            }
        }
        else_if = {
            limit = {
                is_defender_of_faith_of_tier = 4
            }
            change_variable = {            
                which = gt_gained_score
                value = 80
            }
        }
        else_if = {
            limit = {
                is_defender_of_faith_of_tier = 3
            }
            change_variable = {            
                which = gt_gained_score
                value = 60
            }
        }
        else_if = {
            limit = {
                is_defender_of_faith_of_tier = 2
            }
            change_variable = {            
                which = gt_gained_score
                value = 40
            }
        }
        else_if = {
            limit = {
                is_defender_of_faith_of_tier = 1
            }
            change_variable = {            
                which = gt_gained_score
                value = 20
            }
        }
        #Age multiplication
        if = {
            limit = {
                current_age = age_of_revolutions
            }
            multiply_variable = {
                which = gt_gained_score
                value = 4
            }
        }
        else_if = {
            limit = {
                current_age = age_of_absolutism
            }
            multiply_variable = {
                which = gt_gained_score
                value = 3
            }
        }
        else_if = {
            limit = {
                current_age = age_of_reformation
            }
            multiply_variable = {
                which = gt_gained_score
                value = 2
            }
        }
        #Transfer score to overlord
        if = {
            limit = {
                is_subject = yes
            }
            export_to_variable = {
                which = gt_liberty_desire
                value = trigger_value:liberty_desire
                who = ROOT
            }
            if = {
                limit = {
                    NOT = {
                        check_variable = {
                            which = gt_liberty_desire
                            value = 50
                        }
                    }
                }
                set_variable = {
                    which = gt_liberty_desire
                    value = 50
                }
            }
            set_variable = {
                which = gt_score_transfer
                which = gt_gained_score
            }
            multiply_variable = {
                which = gt_gained_score
                which = gt_liberty_desire
            }
            divide_variable = {
                which = gt_gained_score
                value = 100
            }
            #round down
            divide_variable = {
                which = gt_gained_score
                value = 1000
            }
            multiply_variable = {
                which = gt_gained_score
                value = 1000
            }
            subtract_variable = {
                which = gt_score_transfer
                which = gt_gained_score
            }
            overlord = {
                set_variable = {
                    which = gt_score_transfer
                    value = 0
                }
                change_variable = {
                    which = gt_score_transfer
                    which = ROOT
                }
                change_variable = {
                    which = gt_score
                    which = gt_score_transfer
                }
            }
        }
        #Add to total score
        change_variable = {
            which = gt_score
            which = gt_gained_score
        }
    }

    option = {
        name = GT_score.3.a
    }
}

    immediate = {
        set_variable = {            
            which = gt_gained_score
            value = 0
        }
        #100 points for owning at least one of the objectives from a Conqueror Objective pair. Owning both objectives in a pair does not provide additional points, but if another Despot owns the other Conqueror Objective from that pair, the value of both objectives becomes 0 pts for Despots as long as two separate despots each own one objective of the pair..
        #Danzig - Novgorod
        if = {
