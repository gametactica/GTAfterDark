        is_part_of_hre = yes
        is_in_trade_league = yes

	                }
                    NOT = { amount = 12 }

                calc_true_if = {
	                all_country = {
                        limit = { is_part_of_hre = yes }
		                has_reform = free_city
	                }
	                NOT = { amount = 15 }
                }

        }

    option =

		add_government_reform = free_city
        capital_scope = {
            add_base_tax = 2
            add_base_production = 2
            add_base_manpower = 2
        }
    }
