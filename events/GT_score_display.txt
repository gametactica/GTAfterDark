#All score display
country_event = {
    id = GT_score_display.2
    title = GT_score_display.2.t
    is_triggered_only = yes
    hidden = yes
    trigger = {
        always = yes
    }

    immediate = {
        set_country_flag = gt_all_score_display
        if = {
            limit = {
                has_country_flag = gt_campaign_nation
                is_variable_equal = {
                    which = gt_campaign_nation_order
                    value = 0
                }
            }
            country_event = {
                id = GT_score_display.3
            }
        }
        else = {
            random_country = {
                limit = {
                    has_country_flag = gt_campaign_nation
                    is_variable_equal = {
                        which = gt_campaign_nation_order
                        value = 0
                    }
                }
                country_event = {
                    id = GT_score_display.3
                }
            }
        }
    }

    option = {
        name = GT_score_display.2.a
    }
}

country_event = {
    id = GT_score_display.3
	title = GT_score_display.3.t
    is_triggered_only = yes
    hidden = yes
    trigger = {
        always = yes
    }

    immediate = {
        if = {
            limit = {
                is_variable_equal = {
                    which = gt_campaign_nation_order
                    value = 14
                }
            }
            every_country = {
                limit = {
                    has_country_flag = gt_all_score_display
                }
                clr_country_flag = gt_all_score_display
                country_event = {
                    id = GT_score_display.4
                }
            }
        }
        else = {
            random_country = {
                limit = {
                    has_country_flag = gt_campaign_nation
                    check_variable = {
                        which = gt_campaign_nation_order
                        which = ROOT
                    }
                    all_country = {
                        OR = {
                            NOT = { has_country_flag = gt_campaign_nation }
                            NOT = {
                                check_variable = {
                                    which = gt_campaign_nation_order
                                    which = ROOT
                                }
                            }   
                            tag = ROOT
                            check_variable = {
                                which = gt_campaign_nation_order
                                which = PREV
                            }
                        }
                    }
                }
                country_event = {
                    id = GT_score_display.3
                }
            }
        }
    }

    option = {
        name = GT_score_display.3.a
    }
}
